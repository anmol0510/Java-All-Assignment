import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class Pizza {
public static void main(String[] args) throws ClassNotFoundException, SQLException {
	//connect with database
			Class.forName("org.postgresql.Driver");
			
			//create the connection
			Connection conn=DriverManager.getConnection("jdbc:postgresql://localhost/mumbai","postgres","12345@");
			
			//create statement
			Statement stmt = conn.createStatement();
			
			ResultSet result = stmt.executeQuery("select * from pizza");
			
			while(result.next()) {
				System.out.println(result.getInt(1)+" "+result.getInt(2));
			}

			
			conn.close();
			System.out.println("fetching values");
}
}
